410    public static function loadConfig(?string $filename = null): array
411    {

414        $configPath = __DIR__ . '/../../';

419        }
420        foreach ($files as $file) {

422                $configFile = realpath($file);

426        }

428        // Parse config file

434            }

437        // Set config value
438        $appliedConfig = [];
