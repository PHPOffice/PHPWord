97     * Create a hashed password that MS Word will be able to work with.
98     *
99     * @see https://blogs.msdn.microsoft.com/vsod/2010/04/05/how-to-set-the-editing-restrictions-in-word-using-open-xml-sdk-2-0/
100     *
101     * @param string $password
102     * @param string $algorithmName
103     * @param string $salt
104     * @param int $spinCount
105     *
106     * @return string
107     */
108    public static function hashPassword($password, $algorithmName = self::ALGORITHM_SHA_1, $salt = null, $spinCount = 10000)
109    {
110        $origEncoding = mb_internal_encoding();
111        mb_internal_encoding('UTF-8');
112
113        $password = mb_substr($password, 0, min(self::$passwordMaxLength, mb_strlen($password)));
114
115        //   Get the single-byte values by iterating through the Unicode characters of the truncated password.
116        //   For each character, if the low byte is not equal to 0, take it. Otherwise, take the high byte.
117        $passUtf8 = mb_convert_encoding($password, 'UCS-2LE', 'UTF-8');
118        $byteChars = [];
119
120        for ($i = 0; $i < mb_strlen($password); ++$i) {
121            $byteChars[$i] = ord(substr($passUtf8, $i * 2, 1));
122
123            if ($byteChars[$i] == 0) {
124                $byteChars[$i] = ord(substr($passUtf8, $i * 2 + 1, 1));
125            }
126        }
127
128        // build low-order word and hig-order word and combine them
129        $combinedKey = self::buildCombinedKey($byteChars);
130        // build reversed hexadecimal string
131        $hex = str_pad(strtoupper(dechex($combinedKey & 0xFFFFFFFF)), 8, '0', \STR_PAD_LEFT);
132        $reversedHex = $hex[6] . $hex[7] . $hex[4] . $hex[5] . $hex[2] . $hex[3] . $hex[0] . $hex[1];
133
134        $generatedKey = mb_convert_encoding($reversedHex, 'UCS-2LE', 'UTF-8');
135
136        // Implementation Notes List:
137        //   Word requires that the initial hash of the password with the salt not be considered in the count.
138        //   The initial hash of salt + key is not included in the iteration count.
139        $algorithm = self::getAlgorithm($algorithmName);
140        $generatedKey = hash($algorithm, $salt . $generatedKey, true);
141
142        for ($i = 0; $i < $spinCount; ++$i) {
143            $generatedKey = hash($algorithm, $generatedKey . pack('CCCC', $i, $i >> 8, $i >> 16, $i >> 24), true);
144        }
145        $generatedKey = base64_encode($generatedKey);
146
147        mb_internal_encoding($origEncoding);
148
149        return $generatedKey;
150    }

181    /**
182     * Build combined key from low-order word and high-order word.
183     *
184     * @param array $byteChars byte array representation of password
185     *
186     * @return int
187     */
188    private static function buildCombinedKey($byteChars)
189    {
190        $byteCharsLength = count($byteChars);
191        // Compute the high-order word
192        // Initialize from the initial code array (see above), depending on the passwords length.
193        $highOrderWord = self::$initialCodeArray[$byteCharsLength - 1];
194
195        // For each character in the password:
196        //   For every bit in the character, starting with the least significant and progressing to (but excluding)
197        //   the most significant, if the bit is set, XOR the keys high-order word with the corresponding word from
198        //   the Encryption Matrix
199        for ($i = 0; $i < $byteCharsLength; ++$i) {
200            $tmp = self::$passwordMaxLength - $byteCharsLength + $i;
201            $matrixRow = self::$encryptionMatrix[$tmp];
202            for ($intBit = 0; $intBit < 7; ++$intBit) {
203                if (($byteChars[$i] & (0x0001 << $intBit)) != 0) {
204                    $highOrderWord = ($highOrderWord ^ $matrixRow[$intBit]);
205                }
206            }
207        }
208
209        // Compute low-order word
210        // Initialize with 0
211        $lowOrderWord = 0;
212        // For each character in the password, going backwards
213        for ($i = $byteCharsLength - 1; $i >= 0; --$i) {
214            // low-order word = (((low-order word SHR 14) AND 0x0001) OR (low-order word SHL 1) AND 0x7FFF)) XOR character
215            $lowOrderWord = (((($lowOrderWord >> 14) & 0x0001) | (($lowOrderWord << 1) & 0x7FFF)) ^ $byteChars[$i]);
216        }
217        // Lastly, low-order word = (((low-order word SHR 14) AND 0x0001) OR (low-order word SHL 1) AND 0x7FFF)) XOR strPassword length XOR 0xCE4B.
218        $lowOrderWord = (((($lowOrderWord >> 14) & 0x0001) | (($lowOrderWord << 1) & 0x7FFF)) ^ $byteCharsLength ^ 0xCE4B);
219
220        // Combine the Low and High Order Word
221        return self::int32(($highOrderWord << 16) + $lowOrderWord);
222    }
